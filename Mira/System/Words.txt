(splitHalve) "splits list into halves"
(
  split
  (itemsCount) put
  itemsCount 2 divide (itemsCount1) put
  itemsCount itemsCount1 subtract (itemsCount2) put
  itemsCount1 join (result1) put
  itemsCount2 join
  result1
)
word
